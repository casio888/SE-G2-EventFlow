@startuml
start

.Veranstalter mvp flow
[plantuml, Veranstalter-mvp, svg]
----
@startuml
start

if (Account vorhanden) then (ja)
:anmelden;
else (nein)
:registrieren;
endif

if (events vorhanden) then (ja)
:Event bearbeiten oder löschen;
else (nein)
:Event anlegen;
endif

if (Einreichungen vorhanden?) then (ja)
  :Eingegangene Einreichungen ansehen;
  :Einreichung bewerten;

  if (Einreichung geeignet?) then (ja)
    :Einreichung zusagen;
    :Einreicher einem Zeitslot zuweisen;
  else (nein)
    :Einreichung absagen;
  endif

else (nein)
  :Auf Einreichungen warten;
endif
stop
@enduml

----

* nur auf event bewerben ohne zeitslot Logik (z.b. nur btn auf event bewerben)
